00030 /**
00031  * Provides CDN services for ObjectStorage.
00032  *
00033  * CDN stands for "Content Distribution Network." It provides distributed
00034  * caching in the cloud. When a Container is CDN-enabled, objects will be
00035  * stored not @em just in the ObjectStorage, but temporary cached copies
00036  * will be stored on servers around the world.
00037  *
00038  * @attention
00039  * Caches are not protected by authentication. If you store an object in
00040  * a CDN, cached versions of that object are publically accessible. Setting
00041  * an ACL will have very little impact on this.
00042  *
00043  * CDN is an HPCloud extension service, and is not presently part of OpenStack
00044  * proper. The current REST API documentation can be found at
00045  * http://api-docs.hpcloud.com/
00046  *
00047  * <b>Usage</b>
00048  *
00049  * The CDN service functions as an <i>add-on</i> to ObjectStorage. It adds a
00050  * caching layer. So terms used here, such as Container and Object, refer
00051  * to the ObjectStorage items.
00052  *
00053  * For the most part, CDN operates on the Container level. You can choose to
00054  * tell the CDN about a particular Container in ObjectStorage, and it will
00055  * cache items in that container.
00056  *
00057  * The CDN service keeps a list of ObjectStorage Containers that it knows
00058  * about. CDN does not automatically discover ObjectStorage Containers; you
00059  * must tell CDN about the ObjectStorage instances you want it to know
00060  * about. This is done using CDN::enable().
00061  *
00062  * Once the CDN service knows about an ObjectStorage Container, it will
00063  * begin caching objects in that container.
00064  *
00065  * This library gives the the ability to do the following:
00066  *
00067  * - List the containers that CDN knows about: CDN::containers()
00068  * - Retrieve the CDN properties for a particular Container: CDN::container().
00069  * - Add and enable containers: CDN::enable().
00070  * - Remove a Container from CDN: CDN::delete().
00071  * - Modify the caching properties of a Container: CDN::update().
00072  * - Temporarily enable or disable caching for a container with
00073  *   CDN::update().
00074  *
00075  * <b>Example</b>
00076  *
00077  * @code
00078  * <?php
00079  * // Authentication info:
00080  * $endpoint = 'https://auth.example.com';
00081  * $username = 'butcher@hp.com';
00082  * $password = 'secret';
00083  * $tenantId = '123456789';
00084  *
00085  * // First we need to authenticate:
00086  * $identity = new \HPCloud\Services\IdentityServices($endpoint);
00087  * $token = $identity->authenticateAsUser($username, $password, $tenantId);
00088  *
00089  * // Get the service catalog. We will try to have CDN build itself from
00090  * // the service catalog.
00091  * $catalog = $identity->serviceCatalog();
00092  *
00093  * // Get a new CDN instance:
00094  * $cdn = CDN::newFromServiceCatalog($catalog, $token);
00095  *
00096  * // Add a container to CDN; set cache lifetime to an hour:
00097  * $cdn->enable('myContainer', 3600);
00098  *
00099  * // Get a list of all containers that CDN knows about,
00100  * // and print cache lifetime for each:
00101  * foreach ($cdn->containers() as $container) {
00102  *   print $container['name'] . ':' . $container['ttl'] . PHP_EOL;
00103  * }
00104  *
00105  * // Change the cache lifetime on our container
00106  * $cdn->update('myContainer', array('ttl' => 7200));
00107  *
00108  * // Temporarily stop the container from caching:
00109  * $cdn->update('myContainer', array('cdn_enabled' => FALSE);
00110  *
00111  * //This can be re-enabled again:
00112  * $cdn->update('myContainer', array('cdn_enabled' => TRUE);
00113  *
00114  * // If we no longer want this Container in CDN, we
00115  * // should delete it, not just disable it:
00116  * $cdn->delete('myContainer');
00117  *
00118  * ?>
00119  * @endcode
00120  */

00350   /**
00351    * Enable a container.
00352    *
00353    * This adds the container to the CDN service and turns on caching.
00354    *
00355    * In the CDN API, there are two meanings for the term "enable":
00356    *
00357    * 1. To "CDN-enable" a container means to add that container to the CDN
00358    * service. There is no "CDN-disable".
00359    * 2. To "enable" a container means to cache that container's
00360    * content in a publically available CDN server. There is also a
00361    * way to "disable" in this sense -- which blocks a container from 
00362    * caching.
00363    *
00364    * This method does the first -- it adds a container to the CDN
00365    * service. It so happens that adding a container also enables (in the
00366    * second sense) the
00367    * container. (This is a feature of the remote service, not the API).
00368    *
00369    * Enabling and disabling (in the second sense) are considered temporary operations
00370    * to switch on and off caching on a particular container. Both of
00371    * these operations are done with the update() method.
00372    *
00373    * The endpoint is supposed to return different results based on the above;
00374    * accordingly this method should return TRUE if the container was added
00375    * to the list, and FALSE if it was already there. HOWEVER, in some versions
00376    * of the CDN service the endpoint returns the same code for both operations,
00377    * so the result cannot be relied upon.
00378    *
00379    * @param string $name
00380    *   The name of the container.
00381    * @param int $ttl
00382    *   Time to live.
00383    *   The number of seconds an object may stay in the cache. This is the
00384    *   maximum amount of time. There is, however, no assurance that the object
00385    *   will remain for the full TTL. 15 minutes is the minimum time. Five years
00386    *   is the max.
00387    * @param boolean $created
00388    *   If this is passed, then its value will be set to TRUE if the
00389    *   container was created in the CDN, or FALSE if the container
00390    *   already existed in CDN.
00391    * @return string
00392    *   TRUE if the container was created, FALSE if the container was already
00393    *   added to the CDN (and thus nothing happened).
00394    * @throws HPCloud::Exception
00395    *   Several HTTP-level exceptions can be thrown.
00396    * @see http://api-docs.hpcloud.com/hpcloud-cdn-storage/1.0/content/cdn-enable-container.html
00397    */
00398   public function enable($name, $ttl = NULL, &$created = FALSE) {
00399     $headers = array();
00400     if (!empty($ttl)) {
00401       $headers['X-TTL'] = (int) $ttl;
00402     }
00403     $res = $this->modifyContainer($name, 'PUT', $headers);
00404     $created = $res->status() == 201;
00405 
00406     $url = $res->header('X-Cdn-Uri', 'UNKNOWN');
00407     return $url;
00408   }

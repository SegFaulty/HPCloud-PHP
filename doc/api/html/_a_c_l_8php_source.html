00030 /**
00031  * Access control list for object storage.
00032  *
00033  * @b EXPERIMENTAL: This is bassed on a feature of Swift that is likely to
00034  * change. Most of this is based on undocmented features of the API
00035  * discovered both in the Python docs and in discussions by various
00036  * members of the OpenStack community.
00037  *
00038  * Swift access control rules are broken into two permissions: READ and
00039  * WRITE. Read permissions grant the user the ability to access the file
00040  * (using verbs like GET and HEAD), while WRITE permissions allow any
00041  * modification operation. WRITE does not imply READ.
00042  *
00043  * In the current implementation of Swift, access can be assigned based
00044  * on two different factors:
00045  *
00046  * - @b Accounts: Access can be granted to specific accounts, and within
00047  *   those accounts, can be further specified to specific users. See the
00048  *   addAccount() method for details on this.
00049  * - @b Referrers: Access can be granted based on host names or host name
00050  *   patterns. For example, only subdomains of <tt>*.example.com</tt> may be
00051  *   granted READ access to a particular object.
00052  *
00053  * ACLs are transmitted within the HTTP headers for an object or
00054  * container. Two headers are used: @c X-Container-Read for READ rules, and
00055  * @c X-Container-Write for WRITE rules. Each header may have a chain of
00056  * rules.
00057  *
00058  * @b Examples
00059  *
00060  * For most casual cases, only the static constructor functions are
00061  * used. For example, an ACL that does not grant any public access can
00062  * be created with a single call:
00063  *
00064  * @code
00065  * <?php
00066  * $acl = ACL::makeNonPublic();
00067  * ?>
00068  * @endcode
00069  *
00070  * Public read access is granted like this:
00071  *
00072  * @code
00073  * <?php
00074  * $acl = ACL::makePublic();
00075  * ?>
00076  * @endcode
00077  *
00078  * (Note that in both cases, what is returned is an instance of an ACL with
00079  * all of the necessary configuration done.)
00080  *
00081  * Sometimes you will need more sophisticated access control rules. The
00082  * following grants READ access to anyone coming from an @c example.com
00083  * domain, but grants WRITE access only to the account @c admins:
00084  *
00085  * @code
00086  * <?php
00087  * $acl = new ACL();
00088  *
00089  * // Grant READ to example.com users.
00090  * $acl->addReferrer(ACL::READ, '*.example.com');
00091  *
00092  * // Allow only people in the account 'admins' access to
00093  * // write.
00094  * $acl->addAccount(ACL::WRITE, 'admins');
00095  *
00096  * // Allow example.com users to view the container
00097  * // listings:
00098  * $acl->allowListings();
00099  *
00100  * ?>
00101  * @endcode
00102  *
00103  *
00104  * Notes
00105  *
00106  * - The current implementation does not do any validation of rules.
00107  *   This will likely change in the future.
00108  * - There is discussion in OpenStack about providing a different or
00109  *   drastically improved ACL mechanism. This class would then be
00110  *   replaced by a new mechanism.
00111  *
00112  * For a detailed description of the rules for ACL creation,
00113  * see http://swift.openstack.org/misc.html#acls
00114  */

00236    *
00237    * This attempts to parse an ACL rule. It is not particularly
00238    * fault-tolerant.
00239    *
00240    * @param int $perm

00248     // This regular expression generates the following:
00249     //
00250     // array(
00251     //   0 => ENTIRE RULE
00252     //   1 => WHOLE EXPRESSION, no whitespace
00253     //   2 => domain compontent
00254     //   3 => 'rlistings', set if .rincludes is the directive
00255     //   4 => account name
00256     //   5 => :username
00257     //   6 => username

00259     $exp = '/^\s*(.r:([a-zA-Z0-9\*\-\.]+)|\.(rlistings)|([a-zA-Z0-9]+)(\:([a-zA-Z0-9]+))?)\s*$/';
00260 
00261     $matches = array();
00262     preg_match($exp, $rule, $matches);
00263 
00264     $entry = array('mask' => $perm);
00265     if (!empty($matches[2])) {
00266       $entry['host'] = $matches[2];
00267     }
00268     elseif (!empty($matches[3])) {
00269       $entry['rlistings'] = TRUE;
00270     }

87  public function create($name, $flavor = 'medium', $typeSpec = NULL) {
88  // Set type spec. As of the initial release of DBaaS, the only support
89  // type is mysql 5.5.
90  if (empty($typeSpec)) {
91  $typeSpec = array(
92  'name' => 'mysql',
93  'version' => '5.5',
94  );
95  }
96  $json = array(
97  'instance' => array(
98  'name' => $name,
99  'flavorRef' => $flavor,
100  'dbtype' => $typeSpec,
101  ),
102  );
103  $url = $this->url . '/instances';
104  $postData = json_encode($json);
105  //fwrite(STDOUT, "POST DATA: $postData\n");
106  $length = strlen($postData);
107  $headers = $this->headers(array(
108  'Accept' => 'application/json',
109  'Content-Length' => $length,
110  'Content-Type' => 'application/json',
111  ));
112  $res = $this->client->doRequest($url, 'POST', $headers, $postData);
113 
114  $results = json_decode($res->content(), TRUE);
